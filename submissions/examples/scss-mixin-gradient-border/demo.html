2. SCSS Architecture Integration

Learn how the gradient border rotating mixin uses pseudo-element overlays, masking vectors, and keyframe loops:

Mixin API Definition (scss/_mixins.scss) 
@mixin gradient-border-animation(
  $border-width: 2px,
  $gradient: conic-gradient(#6366f1, #10b981, #3b82f6, #6366f1),
  $border-radius: 12px,
  $animation-duration: 4s,
  $background-color: var(--ease-color-surface, #ffffff)
) {
  position: relative;
  overflow: hidden;
  border-radius: $border-radius;
  z-index: 0;

  // Background rotating layer
  &::before {
    content: '';
    position: absolute;
    z-index: -2;
    top: -50%; left: -50%;
    width: 200%; height: 200%;
    background: $gradient;
    animation: ease-kf-gradient-border-rotate $animation-duration linear infinite;
  }

  // Foreground masking layer
  &::after {
    content: '';
    position: absolute;
    z-index: -1;
    top: $border-width; left: $border-width;
    right: $border-width; bottom: $border-width;
    background-color: $background-color;
    border-radius: calc($border-radius - $border-width);
  }
}
Applying the Mixin (style.scss) 
@use '../../../scss/mixins';

.gradient-card-thick {
  // Setup thick custom rotating colors
  @include mixins.gradient-border-animation(
    $border-width: 4px,
    $gradient: conic-gradient(#f59e0b, #ef4444, #f59e0b),
    $animation-duration: 8s,
    $background-color: #111827
  );
}
